1  module clock_generator  
     2  (
     3      input           i2c_core_clk_i          ,   // i2c core clock
     4      input           clk_en_i                ,   // enbale clock to scl
     5      input           reset_ni                ,   // negetive reset signal from MCU
     6      input	[7:0]   prescale_i              ,   // the value used to divide i2c_clock_core to scl line
     7      output          i2c_scl_o                   // scl output
     8  );
     9  
     10      reg             i2c_clk                 ;
     11      reg     [7:0]   counter                 ;
     12  
     13  	assign	i2c_scl_o	=	i2c_clk		    ;	
     14  
     15      // divide i2c core clock to i2c scl output
     16      always @(posedge    i2c_core_clk_i,     negedge   reset_ni) begin
     17  
     18          if (~reset_ni) begin
     19              i2c_clk         <=      1       ;
     20              counter         <=      0       ;
     21          end
     22  
     23          else    begin
     24              if (clk_en_i) begin
     25                  if ( counter == (prescale_i / 2) - 1 ) begin
     26                      i2c_clk     <=   ~i2c_clk    ;
     27                      counter     <=      0        ;
     28                  end 
     29                  else begin
     30                      counter     <=  counter + 1  ;   
     31                  end
     32              end
     33              else begin
     34                  i2c_clk         <=      1        ;
     35              end
     36  
     37          end
     38  
     39      end
     40  
     41  endmodule
