1  module i2c_master_fsm (
     2      input           enable_i            ,   // enable signal from MCU
     3      input           reset_ni            ,   // reset negative signal from MCU
     4      input           repeat_start_i      ,   // repeat start signal from MCU
     5      input           rw_i                ,   // bit 1 is read - 0 is write
     6      input           full_i              ,   // RX_FIFO buffer is full
     7      input           empty_i             ,   // TX_FIFO buffer is empty
     8      input           i2c_core_clk_i      ,   // i2c core clock
     9      input           i2c_sda_i           ,   // i2c sda feedback to FSM
     10      input           i2c_scl_i           ,   // i2c scl feedback to FSM
     11  
     12      output          w_fifo_en_o         ,   //  enable write data into fifo memory
     13      output          r_fifo_en_o         ,   //  enable read data from fifo memory
     14  
     15      output	    reg             sda_low_en_o        ,   // when = 1 enable sda down 0
     16      output	    reg             clk_en_o            ,   // enbale to generator clk
     17      output	    reg             write_data_en_o     ,   // enable write data on sda
     18      output	    reg             write_addr_en_o     ,   // enable write address of slave on sda
     19      output	    reg             receive_data_en_o   ,   // enable receive data from sda
     20      output              [3:0]   count_bit_o         ,   // count bit data from 7 down to 0
     21      output	    reg             i2c_sda_en_o        ,   // allow impact to sda
     22      output	    reg             i2c_scl_en_o        ,   // allow impact to scl 
     23      output                      start_done_o        ,   // state-start done
     24      output                      reset_done_o            
     25  );
     26  
     27      // State
     28      parameter   IDLE            =   4'b0000   ;
     29      parameter   START           =   4'b0001   ;
     30      parameter   ADDRESS         =   4'b0010   ;
     31      parameter   READ_ACK        =   4'b0011   ;
     32      parameter   WRITE_DATA      =   4'b0100   ;
     33      parameter   READ_LATER_ACK  =   4'b0101   ;
     34  
     35      parameter   READ_DATA       =   4'b0110   ;
     36      parameter   WRITE_ACK       =   4'b0111   ;
     37  
     38      parameter   REPEAT_START    =   4'b1000   ;
     39      parameter   STOP            =   4'b1001   ;
     40  
     41      // Declare current state, next state
     42      reg     [3:0]       currrent_state              ;
     43      reg     [3:0]       next_sate                   ;
     44  
     45      // Declare count value
     46      //reg     [2:0]       count_clk_core              ;
     47      reg                 confirm                     ;   // when i2c_scl_i from 1 down to 0, confirm = 1 
     48      reg     [3:0]       count_scl_posedge           ;
     49      reg     [3:0]       count_scl_posedge_temp      ;
     50      reg     [3:0]       count_bit                   ;
     51  	reg		read_ack_to_read_done	                ;
     52  	reg		read_ack_to_write_done	                ;
     53      reg     scl_later		                        ;
     54      wire    scl_positive	                        ;
     55  	wire	scl_negative	                        ;
     56  
     57  
     58      // Declare register of ouput
     59      reg                 w_fifo_en                               ;
     60      reg                 r_fifo_en                               ;
     61      reg                 start_done                              ;
     62      reg                 reset_done                              ;
     63  
     64      assign              count_bit_o         =   count_bit       ;
     65      assign              w_fifo_en_o         =   w_fifo_en       ;
     66      assign              r_fifo_en_o         =   r_fifo_en       ;
     67      assign              start_done_o        =   start_done      ;
     68      assign              reset_done_o        =   reset_done      ;
     69  
     70      // Current State register logic
     71      always @ (posedge i2c_core_clk_i,   negedge reset_ni) begin
     72          if (~reset_ni) begin
     73  
     74              currrent_state      <=  IDLE        ;
     75              reset_done          <=  1           ;
     76  
     77          end
     78  
     79          else begin
     80  
     81              currrent_state  	<=  next_sate   ;
     82              reset_done          <=  0           ;
     83  
     84          end
     85  
     86      end
     87  
     88      // Next state comnibational logic
     89      always @ (*)    begin
     90  
     91          case (currrent_state)
     92  
     93              IDLE    :   begin
     94                  
     95                  if (enable_i ) begin
     96                      next_sate   =   START      ;
     97                  end
     98                  else begin
     99                      next_sate   =   IDLE       ;
     100                  end
     101  
     102              end
     103  
     104  
     105              START   :   begin
     106  
     107  				if (i2c_scl_i == 0) begin                           //  Wait for scl low, and then to next sate
     108  
     109                  	next_sate           =   ADDRESS             ;
     110  
     111                  end
     112  
     113                  else begin
     114  
     115                      next_sate           =   START               ;
     116  
     117                  end
     118  
     119              end
     120  
     121  
     122              ADDRESS :   begin
     123  
     124                  //When the 8 bits data have been transmitted, 
     125                  //wait for the scl line is low and then move to the next state
     126                  if (count_scl_posedge == 8 && scl_negative == 1) begin
     127                      next_sate   =   READ_ACK    ;
     128                  end
     129                  else begin
     130                      next_sate   =   ADDRESS     ;
     131                  end 
     132  
     133              end
     134  
     135              READ_ACK    :   begin
     136  
     137  				// // When have an ACK or NACK, wait for scl is negedge and then to next state
     138                  // if ((scl_negative == 1) && (read_ack_to_read_done == 0 && read_ack_to_write_done == 0)) begin
     139  
     140                  //     next_sate   =   STOP    ;
     141  
     142                  // end
     143  
     144  				// else if ((scl_negative == 1) && (read_ack_to_read_done == 1)) begin
     145  
     146  				// 	next_sate       		=   READ_DATA	;
     147  
     148  				// end
     149  
     150  				// else if ((scl_negative == 1) && (read_ack_to_write_done == 1)) begin
     151  
     152  				// 	next_sate       		=    WRITE_DATA   ;
     153  
     154  				// end				
     155  
     156                  // else begin
     157                  //     next_sate     =       READ_ACK    ;
     158                  // end
     159  
     160                  //modify
     161                  if(scl_negative)
     162                  begin
     163                      if (~read_ack_to_read_done && ~read_ack_to_write_done)
     164                          next_sate = STOP;
     165                      else if (read_ack_to_read_done)
     166                          next_sate = READ_DATA;
     167                      else    
     168                          next_sate = WRITE_DATA;
     169                  end
     170                  else 
     171                      next_sate = READ_ACK;
     172  
     173  
     174              end
     175  
     176              WRITE_DATA  :   begin
     177  
     178                  //When the 8 bits data have been transmitted, 
     179                  //wait for the scl line is high and then move to the next state
     180                  if (count_scl_posedge == 8 && scl_negative == 1) begin
     181  
     182                      next_sate   =   READ_LATER_ACK              ;
     183  
     184                  end 
     185  
     186                  else begin
     187  
     188                      next_sate   =   WRITE_DATA                  ;
     189  
     190                  end
     191  
     192              end
     193  
     194              READ_LATER_ACK  :   begin
     195  
     196  				// if ((scl_negative == 1) && ( (empty_i == 1) || (read_ack_to_write_done == 0 && repeat_start_i == 0) ) ) begin    // If slave send NACK and has no repeart start, or TX_FIFO empty -> STOP
     197  
     198  				// 	next_sate	=	STOP	                    ;
     199  
     200  				// end
     201  
     202                  // else if (scl_negative && ((read_ack_to_write_done == 0) ) ) begin    //  If slave send NACK and has repeart start -> repeat start
     203                      
     204                  //     next_sate   =   REPEAT_START    ;
     205                  // end
     206  
     207  				// else if ((scl_negative) && (read_ack_to_write_done == 1)) begin		// wait for scl low and then to next state
     208  
     209  				// 	next_sate				=	WRITE_DATA		;
     210  
     211  				// end
     212  		
     213                  // else begin
     214  
     215                  //     next_sate           =       READ_LATER_ACK  ;
     216  
     217                  // end
     218  
     219  
     220                  //modify
     221                  if (scl_negative)
     222                  begin
     223                      if (empty_i || (~read_ack_to_write_done && ~repeat_start_i))
     224                          next_sate = STOP;
     225                      else if (read_ack_to_write_done)
     226                          next_sate = WRITE_DATA;
     227                      else
     228                          next_sate = REPEAT_START;
     229                  end
     230                  else
     231                      next_sate = READ_LATER_ACK;
     232  
     233                  //
     234  
     235              end
     236  
     237              READ_DATA   :   begin
     238  
     239                  if (count_scl_posedge == 8 && scl_negative == 1) begin
     240                      next_sate       =       WRITE_ACK       ;
     241                      
     242                  end 
     243                  else begin
     244                      next_sate       =       READ_DATA       ;
     245                  end
     246  
     247              end
     248  
     249              WRITE_ACK   :   begin
     250  
     251  				if (scl_negative) begin
     252  
     253                  	if (repeat_start_i) begin
     254                      	next_sate       =      REPEAT_START     ; 
     255                  	end 
     256                  	else if (full_i == 0) begin
     257                      	next_sate       =       READ_DATA       ;
     258                  	end
     259                  	else begin
     260                      	next_sate       =       STOP            ;
     261                  	end
     262  
     263  				end
     264  				else begin
     265  					next_sate		=		WRITE_ACK		;
     266  				end
     267  
     268              end
     269  
     270              REPEAT_START    :    begin
     271                  if (scl_positive) begin                                 //  Wait to when scl negative and then to next_state
     272                      next_sate           =       START         ;
     273                  end
     274                  else begin
     275                      next_sate           =       REPEAT_START    ;
     276                  end
     277  
     278              end
     279  
     280              STOP    :   begin
     281  
     282  				if (scl_positive) begin
     283                  	next_sate           	=       IDLE            ;
     284  				end
     285  				else begin
     286  					next_sate				=		STOP			;
     287  				end
     288  
     289              end
     290  
     291  
     292              default: next_sate      	=       IDLE            ;
     293          endcase
     294  
     295      end
     296  
     297      // Output and internal signal combinational logic
     298      always @(*) begin
     299          
     300          case (currrent_state)
     301              
     302              IDLE            :   begin
     303                  //  Output
     304                  clk_en_o            =       0           ;
     305                  sda_low_en_o        =       0           ;
     306                  write_data_en_o     =       0           ;
     307                  write_addr_en_o     =       0           ;
     308                  receive_data_en_o   =       0           ;
     309                  i2c_sda_en_o        =       0           ;
     310                  i2c_scl_en_o        =       0           ;
     311  
     312                  //  Internal signal
     313                  start_done              =   0           ;
     314              end
     315  
     316              //-------------------------------------------------------
     317              START           :   begin
     318                  clk_en_o            =       1           ;
     319                  sda_low_en_o        =       1           ;
     320                  write_data_en_o     =       0           ;
     321                  write_addr_en_o     =       0           ;
     322                  receive_data_en_o   =       0           ;
     323                  i2c_sda_en_o        =       1           ;
     324                  i2c_scl_en_o        =       1           ;
     325                  
     326                  //  Internal signal
     327                  start_done              =   1           ;
     328              end
     329  
     330              //-------------------------------------------------------
     331              ADDRESS         :   begin
     332                  clk_en_o            =       1           ;
     333                  sda_low_en_o        =       0           ;
     334                  write_data_en_o     =       0           ;
     335                  receive_data_en_o   =       0           ;
     336  
     337                  //when scl is low, we can write address of slave on sda line
     338                  if (i2c_scl_i == 0)    
     339                      write_addr_en_o     =       1       ;
     340                  else    
     341                      write_addr_en_o     =       0       ;
     342  
     343  				i2c_sda_en_o    		=       1       ;
     344                  i2c_scl_en_o        	=       1       ;
     345                  
     346                  //  Internal signal
     347                  start_done              =   1           ;
     348              end
     349  
     350              //-------------------------------------------------------
     351              READ_ACK        :   begin
     352                  // Output
     353                  clk_en_o            =       1           ;
     354                  sda_low_en_o        =       0           ;
     355                  write_data_en_o     =       0           ;
     356                  write_addr_en_o     =       0           ;
     357                  receive_data_en_o   =       0           ;
     358                  i2c_sda_en_o        =       0           ;
     359                  i2c_scl_en_o        =       1           ;
     360  
     361                  //Internal signal
     362                  start_done              =   1           ;
     363              end
     364  
     365              //-------------------------------------------------------
     366              WRITE_DATA      :   begin
     367                  clk_en_o            =       1           ;
     368                  sda_low_en_o        =       0           ;
     369                  write_addr_en_o     =       0           ;
     370                  receive_data_en_o   =       0           ;
     371  
     372                  //when scl is low, we can write data on sda line
     373                  if (i2c_scl_i == 0)    
     374                      write_data_en_o     =       1           ;
     375                  else    
     376                      write_data_en_o     =       0           ;
     377  
     378  				i2c_sda_en_o    		=       1           ;
     379                  i2c_scl_en_o        	=       1           ;
     380                                  
     381                  //  Internal signal
     382                  start_done              =   1           ;                
     383              end
     384  
     385              //-------------------------------------------------------
     386              READ_LATER_ACK  :   begin
     387                  clk_en_o            =       1           ;
     388                  sda_low_en_o        =       1           ;
     389                  write_data_en_o     =       0           ;
     390                  write_addr_en_o     =       0           ;
     391                  receive_data_en_o   =       0           ;
     392                  i2c_sda_en_o        =       0           ;
     393                  i2c_scl_en_o        =       1           ;
     394  
     395                  // Internal signal
     396                  start_done              =   1           ;
     397              end
     398  
     399              //-------------------------------------------------------
     400              READ_DATA       :   begin
     401                  clk_en_o            =       1           ;
     402                  sda_low_en_o        =       0           ;
     403                  write_data_en_o     =       0           ;
     404                  write_addr_en_o     =       0           ;
     405  
     406                  //when scl is hight, we can receive data from sda
     407                  if (i2c_scl_i)    
     408                      receive_data_en_o   =       1       ;
     409                  else    
     410                      receive_data_en_o   =       0       ;
     411  
     412                  i2c_sda_en_o        =       0           ;
     413                  i2c_scl_en_o        =       1           ;
     414                                  
     415                  //  Internal signal
     416                  start_done              =   1           ;
     417              end
     418  
     419              //-------------------------------------------------------
     420              WRITE_ACK       :   begin
     421                  clk_en_o            =       1           ;
     422                  sda_low_en_o        =       1           ;
     423                  write_data_en_o     =       0           ;
     424                  write_addr_en_o     =       0           ;
     425                  receive_data_en_o   =       0           ;
     426                  if (full_i)
     427                      i2c_sda_en_o    =       0           ;   //  if RX_FIFO full send NACK to slave
     428                  else
     429                      i2c_sda_en_o    =       1           ;
     430                  i2c_scl_en_o        =       1           ;
     431                                  
     432                  //  Internal signal
     433                  start_done              =   1           ;
     434              end
     435  
     436              //-------------------------------------------------------
     437              REPEAT_START    :   begin
     438                  clk_en_o            =       1           ;
     439  				sda_low_en_o    	=       0           ;
     440                  write_data_en_o     =       0           ;
     441                  write_addr_en_o     =       0           ;
     442                  receive_data_en_o   =       0           ;
     443                  i2c_sda_en_o        =       0           ; // off i2c_sda_en to pull sda upto 1
     444                  i2c_scl_en_o        =       1           ;
     445                  
     446                  //  Internal signal
     447                  start_done              =   1           ;
     448              end
     449  
     450              //-------------------------------------------------------
     451              STOP            :   begin
     452                  clk_en_o            =       1           ;
     453                  if (i2c_sda_i == 1) begin                   //  If sda is high, we will set sda to low
     454                      sda_low_en_o    =       1           ;
     455                  end
     456                  else begin
     457                      sda_low_en_o    =       0           ;
     458                  end
     459                  
     460                  write_data_en_o     =       0           ;
     461                  write_addr_en_o     =       0           ;
     462                  receive_data_en_o   =       0           ;
     463                  i2c_sda_en_o        =       1           ;
     464                  i2c_scl_en_o        =       1           ;
     465                                  
     466                  //  Internal signal
     467                  start_done              =   1           ;
     468              end
     469          endcase
     470  
     471      end

     473      //--------------------------------------------------------------------
     474      // Detect positive scl
     475  
     476      always @(posedge    i2c_core_clk_i, negedge     reset_ni) begin
     477          if (~reset_ni) begin
     478              scl_later   <=  1              ;
     479          end
     480          else begin
     481              scl_later   <=  i2c_scl_i       ;
     482          end
     483      end
     484  
     485      assign  scl_positive = i2c_scl_i & (~scl_later) 	;
     486  	assign 	scl_negative = (~i2c_scl_i) & (scl_later)	;
     487  
     488      // end detect positive scl
     489      //------------------------------------------------------------------------
     490      
     491      // Handle count_scl_posedge
     492      always @(*) begin
     493          if (currrent_state  ==  ADDRESS  || currrent_state == READ_DATA  ||  currrent_state  ==  WRITE_DATA) begin
     494              //count_scl_posedge   =   scl_positive ? (count_scl_posedge + 1) : count_scl_posedge  ;
     495              if (scl_positive) begin
     496                  count_scl_posedge   =   count_scl_posedge   +   1   ;
     497              end
     498              else begin
     499                  count_scl_posedge   =   count_scl_posedge_temp      ;
     500              end
     501  
     502          end
     503          else begin
     504              count_scl_posedge   =   0   ;
     505          end
     506      end
     507  
     508  
     509      // Handle read/write-enbale signal to FIFO and count posedge of clock core
     510      always @ (posedge i2c_core_clk_i	, negedge reset_ni) begin
     511          if (~reset_ni) begin
     512              r_fifo_en               <=  0              	;
     513  			w_fifo_en	            <=	0				;
     514              count_scl_posedge_temp  <=  0               ;
     515              count_bit               <=  0               ;
     516              read_ack_to_read_done	<=	0               ;
     517              read_ack_to_write_done	<=	0               ;
     518              
     519          end
     520          else begin
     521              count_scl_posedge_temp  <=  count_scl_posedge   ;
     522  			// enable read/ write data from/to FIFO when READ_LATER_ACK state and WRITE_ACK state
     523  			// at frist READ_ACK state , data is always valid, do not enable read.
     524              if ((currrent_state == READ_LATER_ACK) && (scl_positive == 1)) begin
     525  				r_fifo_en   <=  1           ;
     526  			end
     527  			else begin
     528  				r_fifo_en	<=	0	        ;
     529  			end
     530  
     531  			//if ((currrent_state == WRITE_ACK) && (scl_positive == 1)) begin
     532              if ((currrent_state == READ_DATA) && (next_sate == WRITE_ACK)) begin   //  enable write data to rx_fifo in only 1 clk core
     533  				w_fifo_en   <=  1           ;
     534  			end
     535  			else begin
     536  				w_fifo_en	<=	0	        ;
     537  			end
     538  
     539              // Handle count_bit to transfer to datapath
     540              if (currrent_state  ==  ADDRESS  || currrent_state  ==  WRITE_DATA) begin
     541  
     542              // Each postitive of scl, if count_bit != 0 , reduce count_bit by 1
     543                  count_bit     <=   (count_bit_o != 0 && scl_positive) ? (count_bit - 1) : count_bit  ;
     544  
     545              end
     546  
     547              else if (currrent_state == READ_DATA) begin
     548  
     549                  // Each negative of scl, if count_bit != 0 , reduce count_bit by 1
     550                  count_bit     <=   (count_bit_o != 0 && scl_negative) ? (count_bit - 1) : count_bit   ;
     551  
     552              end
     553  
     554              else begin
     555                  count_bit     <=   7           ;
     556              end
     557  
     558              // Handle internal signal : read_ack_to_read/write_done
     559              if ((currrent_state == READ_ACK) || (currrent_state == READ_LATER_ACK)) begin
     560                  if (scl_positive == 1 && i2c_sda_i == 0 && rw_i == 1 && full_i == 0) begin    
     561  
     562                      read_ack_to_read_done	<=	1         ; 
     563                      read_ack_to_write_done	<=	0         ;
     564  
     565                  end
     566  
     567                  else if (scl_positive == 1 && i2c_sda_i == 0 && rw_i == 0 && empty_i == 0) begin
     568  
     569                      read_ack_to_read_done	<=	0        ;
     570                      read_ack_to_write_done	<=	1        ;
     571  
     572                  end
     573              end
     574              else begin
     575                  read_ack_to_read_done	<=	0        ;
     576                  read_ack_to_write_done	<=	0        ;
     577              end
     578  
     579          end
     580  
     581      end
